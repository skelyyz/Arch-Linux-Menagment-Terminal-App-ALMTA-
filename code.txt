#!/bin/bash

LOGFILE="$HOME/arch-management.log"

# Colors
RED="\e[31m"; GREEN="\e[32m"; YELLOW="\e[33m"; CYAN="\e[36m"; RESET="\e[0m"

log() { echo -e "${CYAN}==>${RESET} $1" | tee -a "$LOGFILE"; }
ok() { echo -e "${GREEN}[OK]${RESET} $1" | tee -a "$LOGFILE"; }
warn() { echo -e "${YELLOW}[WARN]${RESET} $1" | tee -a "$LOGFILE"; }

# --------------------
# SYSTEM MAINTENANCE FUNCTIONS
# --------------------

pre_checks() {
    log "Running pre-checks..."
    ping -c 1 archlinux.org &>/dev/null || warn "No internet connection?"
    if command -v checkupdates >/dev/null; then
        updates=$(checkupdates 2>/dev/null | wc -l)
        log "$updates packages pending update"
    fi
}

snapshot() {
    if command -v btrfs >/dev/null && mount | grep -q "on / type btrfs"; then
        read -rp "Create Btrfs snapshot? (y/N): " s
        if [[ $s =~ ^[Yy]$ ]]; then
            sudo btrfs subvolume snapshot / /.snapshots/pre-$(date +%F-%H%M)
            ok "Snapshot created"
        fi
    elif command -v timeshift >/dev/null; then
        read -rp "Create Timeshift snapshot? (y/N): " s
        [[ $s =~ ^[Yy]$ ]] && sudo timeshift --create --comments "pre-update"
    fi
}

full_update() {
    log "Updating system..."
    sudo pacman -Syu || return
    command -v paru >/dev/null && paru -Sua
    command -v flatpak >/dev/null && flatpak update -y
    command -v fwupdmgr >/dev/null && sudo fwupdmgr refresh && sudo fwupdmgr update
    ok "System updated"
}

cleanup() {
    log "Cleaning cache..."
    sudo paccache -r
    log "Removing orphans..."
    orphan=$(pacman -Qtdq)
    [ -n "$orphan" ] && sudo pacman -Rns $orphan
    log "Cleaning logs..."
    sudo journalctl --vacuum-time=7d
    ok "Cleanup done"
}

post_checks() {
    log "Checking .pacnew files..."
    if command -v pacdiff >/dev/null; then
        pacdiff --output
    else
        warn "Install pacman-contrib for pacdiff"
    fi

    if command -v checkrebuild >/dev/null; then
        log "Checking AUR rebuilds..."
        checkrebuild
    else
        warn "Install rebuild-detector for AUR rebuild checks"
    fi

    if command -v checkservices >/dev/null; then
        log "Services needing restart:"
        checkservices
    else
        warn "Install archlinux-contrib for service restart checks"
    fi
}

health() {
    log "Failed services:"
    systemctl --failed
    log "Disk usage:"
    df -h /
    log "Memory usage:"
    free -h
    log "Pacman cache size:"
    sudo du -sh /var/cache/pacman/pkg 2>/dev/null
}

reboot_check() {
    KERNEL_AFTER=$(uname -r)
    if [ "$KERNEL_BEFORE" != "$KERNEL_AFTER" ]; then
        warn "Kernel updated → reboot recommended"
        read -rp "Reboot now? (y/N): " r
        [[ $r =~ ^[Yy]$ ]] && sudo reboot
    else
        ok "No reboot needed"
    fi
}

system_maintenance_menu() {
    while true; do
        clear
        echo -e "${CYAN}===== System Maintenance =====${RESET}"
        echo "1) Full Smart Maintenance"
        echo "2) Update Only"
        echo "3) Cleanup Only"
        echo "4) Health Check"
        echo "5) Back to Main Menu"
        echo "=============================="
        read -rp "Select option: " sys_choice

        KERNEL_BEFORE=$(uname -r)

        case "$sys_choice" in
            1)
                pre_checks
                snapshot
                full_update
                cleanup
                post_checks
                reboot_check
                ;;
            2)
                pre_checks
                full_update
                reboot_check
                ;;
            3)
                cleanup
                ;;
            4)
                health
                ;;
            5)
                break
                ;;
            *)
                warn "Invalid option. Choose 1-5."
                ;;
        esac

        echo
        read -rp "Press Enter to continue..."
    done
}

# --------------------
# NETWORK & INTERNET MANAGEMENT
# --------------------

network_menu() {
    while true; do
        clear
        echo -e "${CYAN}===== Network & Internet Management =====${RESET}"
        echo "1) Show network status"
        echo "2) Restart network service"
        echo "3) List available Wi-Fi networks"
        echo "4) Connect to Wi-Fi"
        echo "5) Ping test"
        echo "6) Show public IP"
        echo "7) Back to main menu"
        echo "========================================="
        read -rp "Select option: " net_choice

        case "$net_choice" in
            1)
                log "Network status:"
                nmcli general status
                nmcli device status
                read -rp "Press Enter to continue..."
                ;;
            2)
                read -rp "Restart NetworkManager service? (y/N): " r
                if [[ $r =~ ^[Yy]$ ]]; then
                    sudo systemctl restart NetworkManager
                    ok "NetworkManager restarted"
                fi
                read -rp "Press Enter to continue..."
                ;;
            3)
                log "Scanning Wi-Fi networks..."
                nmcli device wifi list
                read -rp "Press Enter to continue..."
                ;;
            4)
                read -rp "SSID: " ssid
                read -rsp "Password: " pass
                echo
                nmcli device wifi connect "$ssid" password "$pass"
                read -rp "Press Enter to continue..."
                ;;
            5)
                read -rp "Ping host (default: 8.8.8.8): " host
                host=${host:-8.8.8.8}
                ping -c 4 "$host"
                read -rp "Press Enter to continue..."
                ;;
            6)
                log "Public IP:"
                curl -s ifconfig.me
                echo
                read -rp "Press Enter to continue..."
                ;;
            7)
                break
                ;;
            *)
                warn "Invalid option. Choose 1-7."
                ;;
        esac
    done
}

# --------------------
# MAIN MENU
# --------------------

while true; do
    clear
    echo -e "${CYAN}========== Arch Management v3 ==========${RESET}"
    echo "1) System Maintenance"
    echo "2) Network & Internet Management"
    echo "3) Exit"
    echo "======================================="
    read -rp "Select option: " choice

    case "$choice" in
        1)
            system_maintenance_menu
            ;;
        2)
            network_menu
            ;;
        3)
            echo "Exiting..."
            break
            ;;
        *)
            warn "Invalid option. Choose 1-3."
            ;;
    esac
done

ok "All tasks completed"
